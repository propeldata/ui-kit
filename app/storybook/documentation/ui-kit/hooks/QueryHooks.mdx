## useCounter

import { useCounter } from "@propeldata/ui-kit"  
import fractionUnicode from 'fraction-unicode'

export default function Fraction(props) {
  const numerator = useCounter({ query, timeZone})
  const denominator = useCounter({ query, timeZone})

  return (<p>{fractionUnicode(numerator.value, denominator.value)}</p>)
